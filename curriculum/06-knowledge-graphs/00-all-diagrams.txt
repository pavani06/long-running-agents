let dcnt = 0;

async function renderDiagram(code, boxId) {
  const box = document.getElementById(boxId);
  if (!box) return;
  try {
    const id = 'md' + (++dcnt);
    const { svg } = await mermaid.render(id, code.trim());
    box.innerHTML = svg;
    const s = box.querySelector('svg');
    if (s) s.style.maxWidth = '100%';
  } catch(e) {
    box.innerHTML = '<div class="mload" style="color:var(--color-text-danger)">Erro: ' + (e.message||'render failed').slice(0,80) + '</div>';
  }
}

function card(d) {
  const mid = 'mx-' + d.id;
  const cid = 'cp-' + d.id;
  const escaped = d.code.replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;');
  return `<div class="dc">
    <div class="dtop">
      <span class="dtag ${d.tc}">${d.num||d.type}</span>
      <div>
        <div class="dtitle">${d.title}</div>
        <div class="ddesc">${d.desc}</div>
      </div>
    </div>
    <div class="dkoda"><i class="ti ti-bolt" aria-hidden="true"></i> ${d.koda}</div>
    <div class="mbox" id="${mid}"><div class="mload">Carregando…</div></div>
    <div class="btns">
      <button class="cbtn" onclick="doCopy('${d.id}',this)"><i class="ti ti-copy" aria-hidden="true"></i> Copiar Mermaid</button>
      <button class="cbtn" onclick="toggleCP('${cid}',this)"><i class="ti ti-code" aria-hidden="true"></i> Ver código</button>
    </div>
    <pre class="cpanel" id="${cid}">${escaped}</pre>
  </div>`;
}

function buildCore() {
  const el = document.getElementById('core-content');
  let html = '';
  CONCEPTS.forEach(c => {
    html += `<div class="cblock">
      <div class="cname">
        <span class="cidx">${c.num}</span>${c.name}
        <span class="cfile"><i class="ti ti-file-text" aria-hidden="true"></i> ${c.file}</span>
      </div>
      <div class="drow">`;
    c.diagrams.forEach(d => {
      const fd = {...d, id:`${c.id}-${d.type}`};
      html += card(fd);
    });
    html += '</div></div>';
  });
  el.innerHTML = html;
}

function buildSection(arr, elId, rowClass='drow') {
  const el = document.getElementById(elId);
  el.innerHTML = `<div class="${rowClass}">${arr.map(card).join('')}</div>`;
}

window.doCopy = async function(id, btn) {
  const all = [
    ...CONCEPTS.flatMap(c => c.diagrams.map(d => ({...d, id:`${c.id}-${d.type}`}))),
    ...MEGA, ...PROB, ...PATS, ...SYS
  ];
  const d = all.find(x => x.id === id);
  if (!d) return;
  try {
    await navigator.clipboard.writeText(d.code.trim());
    btn.innerHTML = '<i class="ti ti-check" aria-hidden="true"></i> Copiado!';
    btn.classList.add('ok');
    setTimeout(() => {
      btn.innerHTML = '<i class="ti ti-copy" aria-hidden="true"></i> Copiar Mermaid';
      btn.classList.remove('ok');
    }, 2000);
  } catch(e) {}
};

window.toggleCP = function(cid, btn) {
  const el = document.getElementById(cid);
  if (!el) return;
  const showing = el.style.display === 'block';
  el.style.display = showing ? 'none' : 'block';
  btn.innerHTML = showing
    ? '<i class="ti ti-code" aria-hidden="true"></i> Ver código'
    : '<i class="ti ti-eye-off" aria-hidden="true"></i> Ocultar código';
};
